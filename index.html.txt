function animateCounter(element, target) {
let current = 0;
const increment = target / 50;
const timer = setInterval(() => {
current += increment;
if (current >= target) {
current = target;
clearInterval(timer);
}
element.textContent = Math.floor(current).toLocaleString() + (target >= 1000 ? '+' : '');
}, 30);
}

statsObserver.unobserve(entry.target);
}
});
